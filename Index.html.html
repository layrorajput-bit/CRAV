// Update Proceed button and badge
function updateCartUI(){
const btn = document.getElementById("proceed-btn");

  ***** badge = document.getElementById("cart-badge");

 badge.style.display="block";

 } else {
 badge.style.display="none";
 }
}

// **** MAIN MENU
function openMenu(){
document.getElementById("splash-ui").style.display="none";
document.getElementById("checkout-ui").style.display="none";
document.getElementById("order-ui").style.display="none";
document.getElementById("main-ui").style.display="block";
 updateCartUI();
}

// BACK TO SPLASH
function goSplash(){
document.getElementById("main-ui").style.display="none";

}

// ADD TO CART
function addToCart(name, price, cardId){
 cart.push({name:name, price:price});
 // Highlight card

  // Animate
 card.style.transform="scale(1.05)";

}

// **** CHECKOUT PAGE
function openCheckout(){

 }

}
